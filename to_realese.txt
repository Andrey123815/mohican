Команды для пользования через терминал:

nginx start server --default
//nginx start server --file-path

nginx set-path config --file_path

nginx create config

nginx stop server --soft (окончание приема сокетов в мастер, обработка запросов в очереди на воркерах)
nginx stop server --critical (немедленный останов сервера с получением сокета с воркера)

nginx reload server (применение конфига)
nginx reload --all_servers (перезагрузка всех серверов -> удаление забаненных апстримов из бан-листа, создание нового обьекта класса NginxServer)

nginx finish server
